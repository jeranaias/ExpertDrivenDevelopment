\begin{frame}[shrink=10]{The Problem: Expertise Without Tools}

\begin{columns}[T]
\column{0.55\textwidth}

\navy{Domain experts understand their problems better than anyone.}\\[6pt]
But they can't build the solutions themselves.

\bigskip

\begin{itemize}
  \item Traditional development: contractors, months of lead time, six-figure budgets
  \item IT backlogs: requests sit for months or years
  \item By the time a tool ships, the problem has changed
  \item The people closest to the problem remain disconnected from the solution
\end{itemize}

\bigskip

{\small\hl{Meanwhile:} 80\% of workers given AI tools quit within weeks (Microsoft). Access without training is worse than no access.}

\column{0.4\textwidth}
\begin{center}
\statbox{80\%}{abandon AI tools\\within weeks}

\end{center}

% Slide 5: The Six Skills

\begin{frame}[shrink=5]{The Six 201-Level Skills}

{\small These are \hl{management skills}, not prompting techniques. Every module maps to at least one.}

\vspace{3pt}

\begin{columns}[T]
\column{0.48\textwidth}

\begin{block}{\faSearch\quad 1. Context Assembly}
{\scriptsize Curate background, constraints, and examples \emph{before} tasking AI. Prevents over-sharing sensitive data.}
\end{block}

\begin{block}{\faCheckCircle\quad 2. Quality Judgment}
{\scriptsize Spot reliable vs.\ unreliable content within the \emph{same} output. The Red Pen Review exercise.}
\end{block}

\begin{block}{\faProjectDiagram\quad 3. Task Decomposition}
{\scriptsize Break work into AI-appropriate chunks. Decide what to delegate vs.\ retain.}
\end{block}

\column{0.48\textwidth}

\begin{block}{\faSync*\quad 4. Iterative Refinement}
{\scriptsize First output is a starting point, not a deliverable. Structured revision passes.}
\end{block}

\begin{block}{\faCogs\quad 5. Workflow Integration}
{\scriptsize Embed AI into recurring processes with documented playbooks. Human/AI labels.}
\end{block}

\begin{block}{\faGlobe\quad 6. Frontier Recognition}
{\scriptsize Know where AI excels and fails for \emph{your} work. Share failure cases across the org.}
\end{block}

\end{columns}

\end{frame}
